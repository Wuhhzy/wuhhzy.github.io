MusicPlayer.lua

-- MusicPlayer: Manages ambiance, music and map chunk loading.
-- This method handles the transitions between areas by fading audio,
-- updating environment settigns and applying new music tied to each region.
local function LoadArea(area)
	Transitioning = true
 
	-- Fade out current music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0}):Play()
	task.wait(2)
 
	-- Apply new area settings
	Music.SoundId = "rbxassetid://" .. area.ID.Value
	Music.TimePosition = area:FindFirstChild("TimePos") and area.TimePos.Value or 0
 
	Ambience(area.Name)
 
	-- Fade in new music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0.5}):Play()
	task.wait(2)
 
	Transitioning = false
end
 
-- This method uses recursion to ensure that the correct area loads if the player enters a new area while still loading the previous one.
-- Fun fact about this code snippet: I made a recursive method without even knowing what recursion was!      
local function RepeatBack()
	repeat wait(.05) until not Transitioning
	if LastTouched ~= CurrentlyPlaying then
		CurrentlyPlaying = LastTouched
		LoadArea(Folder[LastTouched])
		RepeatBack()
	end
end
 
-- This method is an event trigger that, when one of the large music colliders are touched, begins loading the next area or defers to RepeatBack() if it's already transitioning.
child.Touched:Connect(function(hit)
	if hit.Parent == game.Players.LocalPlayer.Character then
		LastTouched = child.Name
 
		if not Transitioning then
			CurrentlyPlaying = child.Name
			LoadArea(child)
		else
			RepeatBack()
		end
	end
end)

Follow.cs

-- This is my simple AI script. Every second, it runs findNearestTorso,
-- which gets every player in the game and sees if they are within a certain distance and has done the most damage to it.
-- It then returns the target, and if it did not return null, begins moving towards them.
-- Looking back, I do notice an issue where if the highest damaging player is out of distance, it won't target other
-- characters, but I am aware of that and would know better now moving forwards.
local isFollow = script.Parent.isFollowing
local botroot = script.Parent.HumanoidRootPart

local dist = 650

function findNearestTorso()
    local torso = nil
    local damagevalue = 0
    for _, value in pairs(script.Parent.Names:GetChildren()) do
        local plr = game.Players:FindFirstChild(value.Name)
        if plr ~= nil then
            local char = plr.Character
            local hum = char:FindFirstChild("Humanoid")
			local root = char:FindFirstChild("HumanoidRootPart")
            if char ~= nil and hum and root then
                if (botroot.Position-root.Position).Magnitude <= dist and value.Value > damagevalue then
                    damagevalue = value.Value
					torso = root
                end
            end
        end
    end
    return torso
end

while true do
    wait(1)
    local target = findNearestTorso(script.Parent.HumanoidRootPart.Position)
    if target ~= nil then
        isFollow.Value = true
        script.Parent.Humanoid:MoveTo(target.Position, target)
    else
        isFollow.Value = false
    end
end

	notice.Parent = player.PlayerGui:WaitForChild("MainInterface")
	notice.LocalScript.Disabled = false
	notice.Text1.Text = "You got "..item.."!"
	delay(5, function()
		if notice ~= nil then
		notice:Destroy()
		end
	end)
	end

local module = {
	-- This is the module that, when an enemy dies, it rewards all players who fought it.
	DropStats = function(mob,stats)
		for _, child in pairs(mob.Names:GetChildren()) do
		local dmg = child.Value
		local mhp = mob.Humanoid.MaxHealth
		local mult = dmg/mhp -- the mult represents the % of damage an individual player dealt to the enemy
		local capmult = 1 -- capmult is a variable that, if the player is overleveled, divides the rewards by half.
		if mult > 1 then mult = 1 end
		local Plr = game.Players:FindFirstChild(child.Name)
		if Plr ~= nil and Plr.Status.Level.Value >= stats.LevelReq then
			if Plr.Status.Level.Value >= stats.LevelCap then
				capmult = 2
			end

			-- We have special passes that players can buy to multiply the amount of EXP and Cash obtained,
			-- the loot variable is used to represent that.
			local loot = 1
			if Plr.Status.TripleLoot.Value == 1 then
					loot = 3
			elseif Plr.Status.DoubleLoot.Value == 1 then
					loot = 2
			end
			Sys.NpcKill(Plr, "EXP", "Cash", math.ceil(((stats.EXP*mult) * (loot+GlobalMultiplier)/capmult)), math.ceil(((stats.Cash*mult) * (loot+GlobalMultiplier)/capmult)))
			if stats.DropXAntibody then Sys.DropXA(Plr, "XAntibody", math.ceil((stats.XAnti) * mult)) end
			if stats.DropOmegaData then Sys.DropOD(Plr, "OmegaData", math.ceil((stats.OD) * mult)) end
			if stats.DropCoin and capmult == 1 and mult >= .25 and stats.CorruptedCoins < 10 then Sys.DropCC(Plr, "CorruptionCoins", stats.CorruptedCoins) end
			if stats.DropCoin and capmult == 1 and mult >= .08 and stats.CorruptedCoins >= 10 then Sys.DropCC(Plr, "CorruptionCoins", stats.CorruptedCoins) end
			if stats.DropItem then
				local reqamnt = 175
				if stats.Drop == "Mephistomon" or stats.Drop == "TitamonBlade" or stats.Drop == "Murmukusmon" or stats.Drop == "BluefireMegi" then reqamnt = 150 end
				if math.random(mult*100,200) >= reqamnt then
					if Plr.Status[stats.Drop].Value < 1 then
						Sys.DropItem(Plr, stats.Drop)
						Notify(Plr, stats.Drop)
					end
				 end
			  end
			end
			end
		end;

// This is a method in an FPS Camera script that controls the recoil when firing your weapon.
// It randomly determines the amount of recoil, multiplied by the guns recoil, and then
// smoothly applies recoil and brings the camera back to its original position.
public void Recoil(float recoilX, float recoilY)
{
    recoilYawOffset += (Random.value - 0.5f) * recoilX;
    recoilPitchOffset -= Mathf.Abs(Random.value - 0.5f) * recoilY;
}
 
// In Update:
float retainedYaw = recoilYawOffset * recoilRetention;
float retainedPitch = recoilPitchOffset * recoilRetention;
