MusicPlayer.lua

  -- MusicPlayer: Manages ambiance, music and map chunk loading.
  -- This method handles the transitions between areas by fading audio,
  -- updating environment settigns and applying new music tied to each region.
  local function LoadArea(area)
	Transitioning = true
 
	-- Fade out current music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0}):Play()
	task.wait(2)
 
	-- Apply new area settings
	Music.SoundId = "rbxassetid://" .. area.ID.Value
	Music.TimePosition = area:FindFirstChild("TimePos") and area.TimePos.Value or 0
 
	Ambience(area.Name)
 
	-- Fade in new music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0.5}):Play()
	task.wait(2)
 
	Transitioning = false
end
 
-- This method uses recursion to ensure that the correct area loads if the player enters a new area while still loading the previous one.
-- Fun fact about this code snippet: I made a recursive method without even knowing what recursion was!      
local function RepeatBack()
	repeat wait(.05) until not Transitioning
	if LastTouched ~= CurrentlyPlaying then
		CurrentlyPlaying = LastTouched
		LoadArea(Folder[LastTouched])
		RepeatBack()
	end
end
 
  -- This method is an event trigger that, when one of the large music colliders are touched, begins loading the next area or defers to RepeatBack() if it's already transitioning.
  child.Touched:Connect(function(hit)
	if hit.Parent == game.Players.LocalPlayer.Character then
		LastTouched = child.Name
 
		if not Transitioning then
			CurrentlyPlaying = child.Name
			LoadArea(child)
		else
			RepeatBack()
		end
	end
end)

// This is a method in an FPS Camera script that controls the recoil when firing your weapon.
// It randomly determines the amount of recoil, multiplied by the guns recoil, and then
// smoothly applies recoil and brings the camera back to its original position.
public void Recoil(float recoilX, float recoilY)
{
    recoilYawOffset += (Random.value - 0.5f) * recoilX;
    recoilPitchOffset -= Mathf.Abs(Random.value - 0.5f) * recoilY;
}
 
// In Update:
float retainedYaw = recoilYawOffset * recoilRetention;
float retainedPitch = recoilPitchOffset * recoilRetention;

// These values are NetworkVariables, and the client's UI automatically updates through
// these subscriptions, which eliminates the need for ClientRPCs nor manual sync calls.
// Additionally, they are unsubscribed in OnNetworkDespawn, which prevents a memory leak.
 
_onMeterChanged  = (_, v) => display.SetMeter(v, Ready.Value);
_onReadyChanged  = (_, v) => display.SetReady(v);
_onHPChanged     = (_, v) => display.SetHP(v, MaxHP.Value);
_onMPChanged     = (_, v) => display.SetMP(v, MaxMP.Value);
_onShieldChanged = (_, v) => display.SetShield(v);
 
Meter.OnValueChanged        += _onMeterChanged;
Ready.OnValueChanged        += _onReadyChanged;
HP.OnValueChanged           += _onHPChanged;
MP.OnValueChanged           += _onMPChanged;
ShieldAmount.OnValueChanged += _onShieldChanged;
