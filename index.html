-- MusicPlayer: Manages ambiance, music and map chunk loading.

  -- This method handles the transitions between areas by fading audio, updating environment settigns and applying new music tied to each region.
  local function LoadArea(area)
	Transitioning = true

	-- Fade out current music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0}):Play()
	task.wait(2)

	-- Apply new area settings
	Music.SoundId = "rbxassetid://" .. area.ID.Value
	Music.TimePosition = area:FindFirstChild("TimePos") and area.TimePos.Value or 0

	Ambience(area.Name)

	-- Fade in new music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0.5}):Play()
	task.wait(2)

	Transitioning = false
end

-- This method uses recursion to ensure that the correct area loads if the player enters a new area while still loading the previous one.
-- Fun fact about this code snippet: I made a recursive method without even knowing what recursion was!      
local function RepeatBack()
	repeat wait(.05) until not Transitioning
	if LastTouched ~= CurrentlyPlaying then
		CurrentlyPlaying = LastTouched
		LoadArea(Folder[LastTouched])
		RepeatBack()
	end
end

  -- This method is an event trigger that, when one of the large music colliders are touched, begins loading the next area or defers to RepeatBack() if it's already transitioning.
  child.Touched:Connect(function(hit)
	if hit.Parent == game.Players.LocalPlayer.Character then
		LastTouched = child.Name

		if not Transitioning then
			CurrentlyPlaying = child.Name
			LoadArea(child)
		else
			RepeatBack()
		end
	end
end)

PlayerController.cs
using UnityEngine;

[RequireComponent(typeof(Rigidbody2D), typeof(Animator))]
public class PlayerController : MonoBehaviour
{
    [Header("Movement")]
    [SerializeField] private float moveSpeed = 7f;
    [SerializeField] private float jumpForce = 12f;

    [Header("Ground Check")]
    [SerializeField] private Transform groundCheck;
    [SerializeField] private LayerMask groundLayer;

    private Rigidbody2D rb;
    private Animator anim;
    private bool isGrounded;
    private float horizontal;

    void Awake()
    {
        rb = GetComponent<Rigidbody2D>();
        anim = GetComponent<Animator>();
    }

    void Update()
    {
        horizontal = Input.GetAxisRaw("Horizontal");
        isGrounded = Physics2D.OverlapCircle(groundCheck.position, 0.15f, groundLayer);

        if (Input.GetButtonDown("Jump") && isGrounded)
            rb.linearVelocity = new Vector2(rb.linearVelocity.x, jumpForce);

        if (Input.GetButtonDown("Fire1"))
            Attack();

        // Flip sprite to face movement direction
        if (horizontal != 0)
            transform.localScale = new Vector3(Mathf.Sign(horizontal), 1f, 1f);

        anim.SetFloat("Speed", Mathf.Abs(horizontal));
        anim.SetBool("IsGrounded", isGrounded);
    }

    void FixedUpdate()
    {
        rb.linearVelocity = new Vector2(horizontal * moveSpeed, rb.linearVelocity.y);
    }

    private void Attack()
    {
        anim.SetTrigger("Attack");
        // Hitbox detection handled via animation event -> OnAttackHit()
    }

    public void OnAttackHit()
    {
        Collider2D[] hits = Physics2D.OverlapCircleAll(transform.position, 1.2f);
        foreach (var hit in hits)
        {
            if (hit.CompareTag("Enemy"))
                hit.GetComponent<EnemyHealth>()?.TakeDamage(10);
        }
    }
}

EnemyHealth.cs
using UnityEngine;

public class EnemyHealth : MonoBehaviour
{
    [Header("Health")]
    [SerializeField] private float maxHealth = 100f;
    private float currentHealth;

    [Header("Dismemberment")]
    [SerializeField] private GameObject[] limbObjects;
    [SerializeField] private float dismemberThreshold = 25f;
    [SerializeField] private GameObject gibPrefab;

    private bool isDead = false;
    private Animator anim;

    void Awake()
    {
        currentHealth = maxHealth;
        anim = GetComponent<Animator>();
    }

    public void TakeDamage(float amount, Vector3 hitPoint, bool isHeadshot = false)
    {
        if (isDead) return;

        float finalDamage = isHeadshot ? amount * 2.5f : amount;
        currentHealth -= finalDamage;

        // Dismember a random limb when a large hit comes in
        if (finalDamage >= dismemberThreshold && limbObjects.Length > 0)
            TriggerDismemberment(hitPoint);

        if (currentHealth <= 0f)
            Die(hitPoint);
        else
            anim.SetTrigger("Hit");
    }

    private void TriggerDismemberment(Vector3 hitPoint)
    {
        int idx = Random.Range(0, limbObjects.Length);
        GameObject limb = limbObjects[idx];
        if (limb == null) return;

        // Detach limb and apply physics impulse
        limb.transform.SetParent(null);
        if (limb.TryGetComponent<Rigidbody>(out var rb))
        {
            rb.isKinematic = false;
            rb.AddExplosionForce(300f, hitPoint, 0.5f);
        }
        limbObjects[idx] = null;
    }

    private void Die(Vector3 hitPoint)
    {
        isDead = true;
        anim.SetTrigger("Die");
        // Spawn gib effect at hit location
        if (gibPrefab != null)
            Instantiate(gibPrefab, hitPoint, Quaternion.identity);
        Destroy(gameObject, 4f);
    }
}

CombatManager.cs
using Unity.Netcode;
using UnityEngine;
using System.Collections;

/// <summary>
/// Active Time Battle system — each combatant fills an ATB gauge.
/// When full, they may act. Server-authoritative.
/// </summary>
public class CombatManager : NetworkBehaviour
{
    [SerializeField] private float atbFillRate = 20f; // units per second

    // NetworkVariable keeps all clients in sync automatically
    private NetworkVariable<float> playerATB = new(0f,
        NetworkVariableReadPermission.Everyone,
        NetworkVariableWritePermission.Server);

    private NetworkVariable<bool> playerReady = new(false,
        NetworkVariableReadPermission.Everyone,
        NetworkVariableWritePermission.Server);

    public override void OnNetworkSpawn()
    {
        if (IsServer)
            StartCoroutine(TickATB());
    }

    private IEnumerator TickATB()
    {
        while (true)
        {
            yield return new WaitForSeconds(0.05f);
            if (!playerReady.Value)
            {
                playerATB.Value = Mathf.Min(playerATB.Value + atbFillRate * 0.05f, 100f);
                if (playerATB.Value >= 100f)
                    playerReady.Value = true;
            }
        }
    }

    // Called by the owning client when they choose an action
    [ServerRpc(RequireOwnership = false)]
    public void SubmitActionServerRpc(ActionData action, ServerRpcParams rpcParams = default)
    {
        if (!playerReady.Value) return;

        ResolveAction(action);

        // Reset ATB after acting
        playerATB.Value = 0f;
        playerReady.Value = false;
    }

    private void ResolveAction(ActionData action)
    {
        // Apply damage, healing, or status effects on the server
        Debug.Log($"[Server] Resolving action: {action.type} for {action.actorId}");
    }
}

[System.Serializable]
public struct ActionData : INetworkSerializable
{
    public ulong actorId;
    public string type;   // "attack", "heal", "skill", etc.
    public int targetId;

    public void NetworkSerialize<T>(BufferSerializer<T> s) where T : IReaderWriter
    {
        s.SerializeValue(ref actorId);
        s.SerializeValue(ref type);
        s.SerializeValue(ref targetId);
    }
}
