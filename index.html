-- MusicPlayer: Manages ambiance and music.
-- This method handles the transitions between areas by fading audio,
-- updating environment settings and applying new music tied to each region.
local function LoadArea(area)
	Transitioning = true
 
	-- Fade out current music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0}):Play()
	task.wait(2)
 
	-- Apply new area settings
	Music.SoundId = "rbxassetid://" .. area.ID.Value
	Music.TimePosition = area:FindFirstChild("TimePos") and area.TimePos.Value or 0
 
	Ambience(area.Name)
 
	-- Fade in new music
	TS:Create(Music, TweenInfo.new(2), {Volume = 0.5}):Play()
	task.wait(2)
 
	Transitioning = false
end
 
-- This method uses recursion to ensure that the correct area loads if the player enters a new area while still loading the previous one.
-- Fun fact about this code snippet: I made a recursive method without even knowing what recursion was!      
local function RepeatBack()
	repeat wait(.05) until not Transitioning
	if LastTouched ~= CurrentlyPlaying then
		CurrentlyPlaying = LastTouched
		LoadArea(Folder[LastTouched])
		RepeatBack()
	end
end
 
-- Triggers area loading on player contact, deferring to RepeatBack() if a transition is already in progress.
child.Touched:Connect(function(hit)
	if hit.Parent == game.Players.LocalPlayer.Character then
		LastTouched = child.Name
 
		if not Transitioning then
			CurrentlyPlaying = child.Name
			LoadArea(child)
		else
			RepeatBack()
		end
	end
end)

-- Targets the highest-damage dealer within range. Known limitation: if that
-- player leaves range, no fallback target is selected. I'd now add a secondary
-- pass for the next closest player.
local isFollow = script.Parent.isFollowing
local botroot = script.Parent.HumanoidRootPart

local dist = 650

function findNearestTorso()
    local torso = nil
    local damagevalue = 0
    for _, value in pairs(script.Parent.Names:GetChildren()) do
        local plr = game.Players:FindFirstChild(value.Name)
        if plr ~= nil then
            local char = plr.Character
            local hum = char:FindFirstChild("Humanoid")
			local root = char:FindFirstChild("HumanoidRootPart")
            if char ~= nil and hum and root then
                if (botroot.Position-root.Position).Magnitude <= dist and value.Value > damagevalue then
                    damagevalue = value.Value
					torso = root
                end
            end
        end
    end
    return torso
end

// Manages chase/attack/idle states by distance threshold.
// Flips sprite scale based on direction rather than using a separate facing variable.
private void FixedUpdate()
{
	if (isDead || isCharging || playerMovement.isDead) return;

	float distance = Vector3.Distance(player.transform.position, transform.position);
	if (distance < 3.5f && !attackDebounce)
	{
		StartCoroutine(chickenAttack());
	}
	else if (distance < 6f)
	{
		dir = (player.transform.position - transform.position).normalized;
		Vector3 scale = transform.localScale;
		scale.x = dir.x > 0 ? -1.6f : 1.6f;
		transform.localScale = scale;
		rb.linearVelocity = new Vector2(dir.x, dir.y) * 2 * Vector2.right;
		animator.SetBool(IsWalking, true);
	}
	else
	{
		rb.linearVelocity = Vector2.zero;
		animator.SetBool(IsWalking, false);
	}
}

// Coroutine allows WaitForSeconds() to sequence the charge,
// lunge, and cooldown as a single readable attack flow with a debounce.
private IEnumerator ChickenAttack()
{
	isCharging = true;
	attackDebounce = true;
	animator.SetBool(IsCharging, true);
	audioSource.PlayOneShot(swordCharge);
	yield return new WaitForSeconds(0.4f);
	if (!isDead)
	{
		audioSource.PlayOneShot(swordSwing);
		chickenHitbox.SetActive(true);
		rb.linearVelocity = new Vector2(dir.x, dir.y) * 13 * Vector2.right;
		animator.SetBool(IsCharging, false);
		animator.SetBool(IsAttacking, true);
		yield return new WaitForSeconds(0.2f);
		isCharging = false;
		chickenHitbox.SetActive(false);
		animator.SetBool(IsAttacking, false);
		yield return new WaitForSeconds(0.6f);
		attackDebounce = false;
	}
}

FPSCamera.cs
private float recoilYawOffset, recoilPitchOffset;
private float recoilYawVelocity, recoilPitchVelocity;
private float recoilRetention = 0.95f;

// Called externally by guns on fire — adds random horizontal
// and upward kick to the recoil offsets.
public void Recoil(float recoilX, float recoilY)
{
    recoilYawOffset += (Random.value - 0.5f) * recoilX;
    recoilPitchOffset -= Mathf.Abs(Random.value - 0.5f) * recoilY;
}

void Update()
{
    // recoilRetention creates a natural decay curve each frame
    // before SmoothDamp closes the remaining gap.
    recoilYawOffset = Mathf.SmoothDamp(recoilYawOffset, recoilYawOffset * recoilRetention, ref recoilYawVelocity, 0.01f);
    recoilPitchOffset = Mathf.SmoothDamp(recoilPitchOffset, recoilPitchOffset * recoilRetention, ref recoilPitchVelocity, 0.01f);

    // Recoil offsets are layered on top of player look input,
    // then the result is smooth-damped for a natural feeling camera.
    float finalYaw = targetYaw + recoilYawOffset;
    float finalPitch = Mathf.Clamp(targetPitch + recoilPitchOffset, MinPitch, MaxPitch);

    yaw = Mathf.SmoothDamp(yaw, finalYaw, ref yawVelocity, 0.1f);
    pitch = Mathf.SmoothDamp(pitch, finalPitch, ref pitchVelocity, 0.1f);

    transform.localRotation = Quaternion.Euler(pitch, yaw, 0f);
}

// Delegates are stored by reference so they can be cleanly unsubscribed
// in OnNetworkDespawn — avoids stale listeners on destroyed objects.
private NetworkVariable<float>.OnValueChangedDelegate _onMeterChanged;
private NetworkVariable<bool>.OnValueChangedDelegate  _onReadyChanged;
private NetworkVariable<int>.OnValueChangedDelegate   _onHPChanged;
private NetworkVariable<int>.OnValueChangedDelegate   _onMPChanged;
private NetworkVariable<int>.OnValueChangedDelegate   _onShieldChanged;

public override void OnNetworkSpawn()
{
    var display = GetComponent<CombatEntityDisplay>();
    if (display != null)
    {
        // Wiring NetworkVariable callbacks to the display means every client's
        // UI reacts to stat changes automatically, without manual ClientRpcs.
        _onMeterChanged  = (_, v) => display.SetMeter(v, Ready.Value);
        _onReadyChanged  = (_, v) => display.SetReady(v);
        _onHPChanged     = (_, v) => display.SetHP(v, MaxHP.Value);
        _onMPChanged     = (_, v) => display.SetMP(v, MaxMP.Value);
        _onShieldChanged = (_, v) => display.SetShield(v);

        Meter.OnValueChanged        += _onMeterChanged;
        Ready.OnValueChanged        += _onReadyChanged;
        HP.OnValueChanged           += _onHPChanged;
        MP.OnValueChanged           += _onMPChanged;
        ShieldAmount.OnValueChanged += _onShieldChanged;
    }
}

public override void OnNetworkDespawn()
{
    var display = GetComponent<CombatEntityDisplay>();
    if (display != null)
    {
        Meter.OnValueChanged        -= _onMeterChanged;
        Ready.OnValueChanged        -= _onReadyChanged;
        HP.OnValueChanged           -= _onHPChanged;
        MP.OnValueChanged           -= _onMPChanged;
        ShieldAmount.OnValueChanged -= _onShieldChanged;
    }
}

// Server-only. Shield absorbs damage first before HP is touched.
// On death, XP is distributed to every connected player.
public virtual void TakeDamage(int damage)
{
    if (!IsServer || !Alive.Value) return;

    if (ShieldAmount.Value > 0)
    {
        int absorbed = Mathf.Min(ShieldAmount.Value, damage);
        ShieldAmount.Value -= absorbed;
        damage             -= absorbed;
    }

    if (damage <= 0) return;

    HP.Value -= damage;
    ShowDamageRpc(damage);

    if (HP.Value <= 0)
    {
        HP.Value = 0; Alive.Value = false;
        Ready.Value = false; Meter.Value = 0f;

        if (IsAI)
        {
            foreach (var clientId in NetworkManager.Singleton.ConnectedClientsIds)
            {
                var p = NetworkManager.Singleton.SpawnManager
                            .GetPlayerNetworkObject(clientId)
                            ?.GetComponent<PlayerProgression>();
                p?.AddXP(XPReward);
            }
        }
    }
}

// Pure static utility — no MonoBehaviour, no networking.
// Converts a budget + enemy pool into a list of sequential waves.
public static class WaveGenerator
{
    private const int WAVE_SIZE_THRESHOLD = 4;

    public static List<List<EnemyEntry>> Generate(
        int budget, List<EnemyEntry> eligibleEnemies, bool largeWave = false)
    {
        if (largeWave) return GenerateLargeWave(budget, eligibleEnemies);

        var flatList = PickEnemiesWithinBudget(budget, eligibleEnemies);
        return SplitIntoWaves(flatList);
    }

    // Large wave modifier bypasses normal budget logic entirely —
    // floods the encounter with as many cheap enemies as the budget allows.
    private static List<List<EnemyEntry>> GenerateLargeWave(int budget, List<EnemyEntry> pool)
    {
        EnemyEntry cheapest = pool[0];
        foreach (var e in pool)
            if (e.Cost < cheapest.Cost) cheapest = e;

        int count = Mathf.Max(1, budget / cheapest.Cost);
        var wave = new List<EnemyEntry>(count);
        for (int i = 0; i < count; i++) wave.Add(cheapest);
        return new List<List<EnemyEntry>> { wave };
    }

    // Greedy random fill: picks enemies at random until the budget is exhausted.
    // Rebuilds the affordable list each iteration to avoid over-spending.
    private static List<EnemyEntry> PickEnemiesWithinBudget(int budget, List<EnemyEntry> pool)
    {
        var result    = new List<EnemyEntry>();
        int remaining = budget;
        int safety    = 60;

        var affordable = new List<EnemyEntry>(pool.FindAll(e => e.Cost <= remaining));

        while (remaining > 0 && affordable.Count > 0 && safety-- > 0)
        {
            var pick = affordable[Random.Range(0, affordable.Count)];
            result.Add(pick);
            remaining -= pick.Cost;

            affordable.Clear();
            foreach (var e in pool)
                if (e.Cost <= remaining) affordable.Add(e);
        }
        return result;
    }

    // Shuffles the flat enemy list, then distributes round-robin into waves
    // so each wave has a roughly equal mix rather than front-loaded enemies.
    private static List<List<EnemyEntry>> SplitIntoWaves(List<EnemyEntry> enemies)
    {
        var waves = new List<List<EnemyEntry>>();
        if (enemies.Count == 0) return waves;

        for (int i = enemies.Count - 1; i > 0; i--)
        {
            int j = Random.Range(0, i + 1);
            (enemies[i], enemies[j]) = (enemies[j], enemies[i]);
        }

        int waveCount = Mathf.Max(1, Mathf.CeilToInt((float)enemies.Count / WAVE_SIZE_THRESHOLD));
        for (int w = 0; w < waveCount; w++) waves.Add(new List<EnemyEntry>());

        for (int i = 0; i < enemies.Count; i++)
            waves[i % waveCount].Add(enemies[i]);

        return waves;
    }
}
DungeonState.cs
// Attempts to resolve the floor vote after each submission.
// Majority wins immediately; if all players vote with no majority,
// the highest vote count wins with a random tiebreak.
private void TryResolveVote()
{
    int playerCount = NetworkManager.Singleton.ConnectedClientsIds.Count;
    int majority    = playerCount / 2 + 1;

    var counts = new Dictionary<int, int>();
    foreach (var v in _floorVotes.Values)
    {
        if (!counts.ContainsKey(v)) counts[v] = 0;
        counts[v]++;
    }

    // Early exit: a majority has been reached
    foreach (var kv in counts)
        if (kv.Value >= majority) { CommitSelection(kv.Key); return; }

    // All players have voted but no majority — pick highest, random on tie
    if (_floorVotes.Count >= playerCount)
    {
        int best = -1, bestCount = -1;
        foreach (var kv in counts)
            if (kv.Value > bestCount) { best = kv.Key; bestCount = kv.Value; }
        CommitSelection(best);
    }
}

    var eligible = new List<EnemyEntry>();
    foreach (var e in Enemies)
        if (currentFloor <= e.IntroducedAtFloor + CullThreshold)
            eligible.Add(e);

    // Fallback: if every enemy was culled, return the most expensive ones
    // so the encounter still feels appropriately threatening.
    if (eligible.Count == 0)
    {
        int maxCost = 0;
        foreach (var e in Enemies) if (e.Cost > maxCost) maxCost = e.Cost;
        foreach (var e in Enemies) if (e.Cost == maxCost) eligible.Add(e);
    }

    return eligible;
}

[Rpc(SendTo.Server)]
public void SetReadyServerRpc(RpcParams rpcParams = default)
{
    if (_departing) return;

    ulong sender = rpcParams.Receive.SenderClientId;
    if (!_readyClients.Add(sender)) return; // deduplicate double-ready calls

    ReadyCount.Value = _readyClients.Count;
    int total        = NetworkManager.ConnectedClientsIds.Count;

    if (_readyClients.Count >= total)
    {
        _departing = true;
        Depart();
    }
    // Majority ready triggers a countdown rather than requiring unanimity,
    // so one slow player can't hold the group hostage indefinitely.
    else if (_readyClients.Count > total / 2 && !CountdownActive.Value)
    {
        StartCoroutine(DepartureCountdown());
    }
}

private IEnumerator DepartureCountdown()
{
    CountdownActive.Value = true;
    Countdown.Value       = countdownSeconds;

    while (Countdown.Value > 0)
    {
        yield return new WaitForSeconds(1f);
        if (_departing) yield break;
        Countdown.Value--;

        if (_readyClients.Count >= NetworkManager.ConnectedClientsIds.Count)
            break;
    }

    if (!_departing) { _departing = true; Depart(); }
}
